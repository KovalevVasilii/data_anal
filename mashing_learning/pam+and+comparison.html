def create_cluster(data_set, medoids,distance):
    clusters=np.zeros(data_set.shape[0])
    for i in range(data_set.shape[0]):
        num_of_cluster=0;
        min=sys.maxsize
        for j in range(medoids.shape[0]):
            Dist=pdist(np.array([data_set[i],medoids[j]]),metric=distance)
            if Dist < min:
                min=Dist
                num_of_cluster=j
        clusters[i]=num_of_cluster
    return clusters

def choose_medoids(cluster,distance):
    min=sys.maxsize
    for i in range(cluster.shape[0]):
        possible_medoid=np.copy(cluster[i])
        Dist=0
        for j in range(cluster.shape[0]):
            Dist=Dist+pdist(np.array([possible_medoid,cluster[j]]),metric=distance)
        if Dist<min:
            min=Dist
            num_of_medoids=i
    return cluster[num_of_medoids]

def Inertia(data_set, clusters, medoids):
    inertia=0
    for i in range(data_set.shape[0]):
        for j in range(medoids.shape[0]):
            if clusters[i]==j:
                inertia+=np.square(pdist(np.array([data_set[i],medoids[j]])))
    return inertia

def correct_cluster(data_set,clusters,num):
    n=0
    cluster=np.zeros(shape=(n, data_set.shape[1]))
    for i in range(data_set.shape[0]):
        if clusters[i]==num:
            n+=1
            cluster_tmp=np.zeros(shape=(n, data_set.shape[1]))
            for j in range(n-1):
                cluster_tmp[j]=np.copy(cluster[j])
            cluster_tmp[n-1]=np.copy(data_set[i])
            cluster=cluster_tmp
    return cluster

def pam(data_set, n, distance,flag):
    size_of_data=data_set.shape[0]
    medoids=np.zeros(shape=(n, data_set.shape[1]))
    for i in range(n):
        medoids[i]=np.copy(data_set[random.randint(0,size_of_data-1)])
        for j in range(i):
            while np.any(medoids[i]==medoids[j]):
                medoids[i]=np.copy(data_set[random.randint(0,size_of_data-1)])
    clusters=np.copy(create_cluster(data_set,medoids,distance))
    break_flag=True
    while (break_flag) :
        break_flag=False
        new_medoids=np.zeros(shape=(n, data_set.shape[1]))
        for i in range(n):
            new_medoids[i]=np.copy(choose_medoids(correct_cluster(data_set,clusters,i),distance))
        clusters=create_cluster(data_set,new_medoids,distance)
        for i in range(n):
            if np.any(medoids[i]!=new_medoids[i]):
                medoids[i]=np.copy(new_medoids[i])
                break_flag=True
        pca = PCA(2) 
        result = pca.fit_transform(data_set) 
        if flag==True:
            plt.figure(figsize=(12,8)) 
            plt.title('pca') 
            plt.scatter(result[:,0], result[:,1], c=clusters) 
            plt.plot() 
            plt.show()
    inertia=Inertia(data_set,clusters,medoids)
    return inertia, clusters, medoids
